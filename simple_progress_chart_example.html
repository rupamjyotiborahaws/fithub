Simple Member Health Progress Chart

Ultra-simple chart function - just pass data and metric names!

Select Metrics:

Super Simple Usage:

// Just import and use!
import { renderMemberProgressChart } from './charts.js';

// Your health data
const healthData = [
  { measure_date: '2024-01-01', weight: 70, bmi: 23.5 },
  { measure_date: '2024-02-01', weight: 69, bmi: 23.1 },
  // ... more data
];

// Render chart with default metrics (weight, bmi)
renderMemberProgressChart('chartCanvas', healthData);

// Or specify custom metrics
renderMemberProgressChart('chartCanvas', healthData, ['weight', 'height', 'bmi']);

Laravel Example:

// Controller
$healthData = MemberHealthTrack::where('user_id', $userId)
    ->orderBy('measure_date')
    ->get();

// View
<canvas id="progressChart"></canvas>

<script type="module">
import { renderMemberProgressChart } from './resources/js/charts.js';

const data = @json($healthData);
renderMemberProgressChart('progressChart', data, ['weight', 'bmi']);
</script>
